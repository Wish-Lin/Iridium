function cubic(inp){
  return inp*inp*inp - 3*inp;
}
clear();                    //清空畫布
setcanvas(600,600,"white"); //設定600x600的白色畫布
ezsetgrid(10,3,"black",1,1);//設定坐標軸
for(var i = -3;i<3;i+=0.1){
line_pp(i,cubic(i),i+0.1,cubic(i+0.1),2,"round","blue");
}
